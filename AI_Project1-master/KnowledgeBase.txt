RULE: 10
IF DEGREE = NO
THEN POSITION = NO

RULE: 20
IF DEGREE = YES
THEN QUALIFY = YES

RULE: 30
IF DEGREE = YES
AND DISCOVERY = YES
THEN POSITION = RESEARCH

RULE: 40
IF QUALIFY = YES
AND GRADE < 3.5 
AND EXPERIENCE >= 2
THEN POSITION = SERVICEENG

RULE: 50
IF QUALIFY = YES
AND GRADE < 3.5
AND EXPERIENCE < 2
THEN POSITION = NO

RULE: 60
IF QUALIFY = YES
AND GRADE >= 3.5
THEN POSITION = PRODUCTENG
